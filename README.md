<!-- JavaScript for Dark/Light Toggle -->
<script>
let isDark = false;

function toggleMode() {
    const icon = document.getElementById('mode-icon');
    const elements = document.querySelectorAll('[style*="background: linear-gradient"]');
    
    isDark = !isDark;
    
    if (isDark) {
        icon.src = "https://api.iconify.design/ri/moon-line.svg?color=%23c9d1d9";
        icon.alt = "Dark mode icon";
        document.body.style.backgroundColor = "#0d1117";
        document.body.style.color = "#c9d1d9";
        
        elements.forEach(el => {
            if (el.style.background.includes('linear-gradient')) {
                el.style.background = el.style.background
                    .replace(/rgba\(.*?0\.1\)/g, 'rgba(255,255,255,0.05)')
                    .replace(/rgba\(.*?0\.2\)/g, 'rgba(255,255,255,0.1)')
                    .replace(/#f8f9fa/g, '#161b22')
                    .replace(/#ffffff/g, '#0d1117');
            }
        });
    } else {
        icon.src = "https://api.iconify.design/ri/sun-line.svg?color=%23667eea";
        icon.alt = "Light mode icon";
        document.body.style.backgroundColor = "";
        document.body.style.color = "";
        
        elements.forEach(el => {
            if (el.style.background.includes('linear-gradient')) {
                el.style.background = el.style.background
                    .replace(/rgba\(.*?0\.05\)/g, 'rgba(102, 126, 234, 0.1)')
                    .replace(/rgba\(.*?0\.1\)/g, 'rgba(102, 126, 234, 0.2)')
                    .replace(/#161b22/g, '#f8f9fa')
                    .replace(/#0d1117/g, '#ffffff');
            }
        });
    }
}

// Initialize based on system preference
if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
    toggleMode(); // Switch to dark mode if system prefers dark
}
</script>
